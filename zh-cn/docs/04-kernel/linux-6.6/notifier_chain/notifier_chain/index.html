              Linux内核的通知链机制

              2024/08/144 分钟阅读 内核基础 内核基础通知链

              目录

              1. 简介

              Linux内核中的notifier_chain是一种用于跨模块通信的机制。当一个模块需要通知其他模块某些事件时，它可以使用notifier_chain将这些事件通知给其他模块。这种机制使得模块之间的耦合度降低，提高了代码的可维护性和可扩展性。

              在Linux内核中，有四种类型的通知链：

              1. 原子通知链适用于对原子性要求高的中断上下文
              2. 阻塞通知链适用于进程上下文的复杂处理
              3. 原始通知链提供了最大的灵活性但需要谨慎使用
              4. SRCU 通知链则在读写锁保护的场景中发挥作用
               1/// include/linux/notifier.h
 2/*
 3 * Notifier chains are of four types:
 4 *
 5 *	Atomic notifier chains: Chain callbacks run in interrupt/atomic
 6 *		context. Callouts are not allowed to block.
 7 *	Blocking notifier chains: Chain callbacks run in process context.
 8 *		Callouts are allowed to block.
 9 *	Raw notifier chains: There are no restrictions on callbacks,
10 *		registration, or unregistration.  All locking and protection
11 *		must be provided by the caller.
12 *	SRCU notifier chains: A variant of blocking notifier chains, with
13 *		the same restrictions.
14 *
15 * atomic_notifier_chain_register() may be called from an atomic context,
16 * but blocking_notifier_chain_register() and srcu_notifier_chain_register()
17 * must be called from a process context.  Ditto for the corresponding
18 * _unregister() routines.
19 *
20 * atomic_notifier_chain_unregister(), blocking_notifier_chain_unregister(),
21 * and srcu_notifier_chain_unregister() _must not_ be called from within
22 * the call chain.
23 *
24 * SRCU notifier chains are an alternative form of blocking notifier chains.
25 * They use SRCU (Sleepable Read-Copy Update) instead of rw-semaphores for
26 * protection of the chain links.  This means there is _very_ low overhead
27 * in srcu_notifier_call_chain(): no cache bounces and no memory barriers.
28 * As compensation, srcu_notifier_chain_unregister() is rather expensive.
29 * SRCU notifier chains should be used when the chain will be called very
30 * often but notifier_blocks will seldom be removed.
31 */
32
33struct notifier_block;
34
35typedef	int (*notifier_fn_t)(struct notifier_block *nb,
36            unsigned long action, void *data);
37
38struct notifier_block {
39    notifier_fn_t notifier_call;
40    struct notifier_block __rcu *next;
41    int priority;
42};
43
44struct atomic_notifier_head {
45    spinlock_t lock;
46    struct notifier_block __rcu *head;
47};
48
49struct blocking_notifier_head {
50    struct rw_semaphore rwsem;
51    struct notifier_block __rcu *head;
52};
53
54struct raw_notifier_head {
55    struct notifier_block __rcu *head;
56};
57
58struct srcu_notifier_head {
59    struct mutex mutex;
60    struct srcu_usage srcuu;
61    struct srcu_struct srcu;
62    struct notifier_block __rcu *head;
63};

              1.1. 原子通知链

              原子通知链用于在中断上下文中或者其他不允许阻塞的场景中进行通知。其特点是通知的处理过程必须是原子的，不能被打断。

              在实现上，原子通知链通过使用原子操作来保证通知的顺序和完整性。当事件发生时，内核会遍历原子通知链上注册的回调函数，并依次执行它们。

              优点：适用于对实时性和原子性要求较高的场景，能够确保通知的快速处理且不会被中断。

              缺点：由于不能阻塞，所以在处理复杂逻辑时可能会受到限制。

              1.2. 阻塞通知链

              阻塞通知链允许在通知处理过程中进行阻塞操作，例如等待资源、睡眠等。

              这种通知链通常在进程上下文使用，因为阻塞操作在中断上下文是不被允许的。当通知被触发时，内核会按照注册的顺序依次调用回调函数，如果某个回调函数进行了阻塞操作，内核会暂停当前回调函数的执行，切换到其他可运行的任务。

              优点：能够处理较为复杂的逻辑，包括需要阻塞等待的情况。

              缺点：可能会导致系统的响应时间变长，特别是在处理耗时的阻塞操作时。

              1.3. 原始通知链

              原始通知链提供了一种最低级别的通知机制，不对通知的执行顺序和并发访问进行任何保证。

              它适用于那些对通知的顺序和并发处理要求不高，或者需要自定义处理方式的场景。

              优点：灵活性高，能够满足一些特殊的需求。

              缺点：由于缺乏保证，使用时需要特别小心，以避免出现竞态条件和不确定的行为。

              1.4. SRCU 通知链

              SRCU 通知链用于在读写锁保护的场景中进行通知。它结合了读锁的并发访问和写锁的同步更新特性。

              在使用 SRCU 通知链时，读端可以并发地访问通知链，而写端在更新通知链时需要获取写锁。

              优点：在保证一定的并发性能的同时，能够安全地进行通知链的更新。

              缺点：实现相对复杂，需要对读写锁和 SRCU 机制有深入的理解。

              2. 原理

              2.1. 注册

              核心函数：notifier_chain_register。

               1/// kernel/notifier.c
 2static int notifier_chain_register(struct notifier_block **nl,
 3                   struct notifier_block *n,
 4                   bool unique_priority)
 5{
 6    while ((*nl) != NULL) {
 7        /// 已经存在，返回错误
 8        if (unlikely((*nl) == n)) {
 9            WARN(1, "notifier callback %ps already registered",
10                 n->notifier_call);
11            return -EEXIST;
12        }
13        /// 优先级高于当前节点，插入到当前节点之前
14        if (n->priority > (*nl)->priority)
15            break;
16        /// 优先级相同，且要求唯一性，返回错误
17        if (n->priority == (*nl)->priority && unique_priority)
18            return -EBUSY;
19        /// 遍历下一个节点
20        nl = &((*nl)->next);
21    }
22    /// 插入新节点
23    n->next = *nl;
24    rcu_assign_pointer(*nl, n);
25
26    trace_notifier_register((void *)n->notifier_call);
27    return 0;
28}

              2.2. 调用

              核心函数：notifier_call_chain。

               1/// kernel/notifier.c
 2/**
 3 * notifier_call_chain - Informs the registered notifiers about an event.
 4 *	@nl:		Pointer to head of the blocking notifier chain
 5 *	@val:		Value passed unmodified to notifier function
 6 *	@v:		Pointer passed unmodified to notifier function
 7 *	@nr_to_call:	Number of notifier functions to be called. Don't care
 8 *			value of this parameter is -1.
 9 *	@nr_calls:	Records the number of notifications sent. Don't care
10 *			value of this field is NULL.
11 *	Return:		notifier_call_chain returns the value returned by the
12 *			last notifier function called.
13 */
14static int notifier_call_chain(struct notifier_block **nl,
15                   unsigned long val, void *v,
16                   int nr_to_call, int *nr_calls)
17{
18    int ret = NOTIFY_DONE;
19    struct notifier_block *nb, *next_nb;
20
21    /// 获取当前链表第一个节点
22    nb = rcu_dereference_raw(*nl);
23
24    while (nb && nr_to_call) {
25        /// 获取下一个节点
26        next_nb = rcu_dereference_raw(nb->next);
27
28#ifdef CONFIG_DEBUG_NOTIFIERS
29        /// ... ..
30#endif
31        trace_notifier_run((void *)nb->notifier_call);
32        /// 调用回调函数
33        ret = nb->notifier_call(nb, val, v);
34
35        if (nr_calls)
36            (*nr_calls)++;  /// 更新调用次数
37
38        /// 判断回调函数是否需要停止调用
39        if (ret & NOTIFY_STOP_MASK)
40            break;
41
42        /// 更新当前节点
43        nb = next_nb;
44        nr_to_call--;
45    }
46    return ret;
47}
48NOKPROBE_SYMBOL(notifier_call_chain);
